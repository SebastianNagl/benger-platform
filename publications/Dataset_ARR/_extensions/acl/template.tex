% Prompt environment for German prompt blocks in the appendix.
% breaklines wraps overlong lines so they fit in the column;
% breakanywhere lets the break fall mid-word when no space is reachable.
% breaksymbolleft/right are emptied so wrapped lines don't show the
% default \hookrightarrow marker (the prompts are documentary, not code).
\DefineVerbatimEnvironment{Prompt}{Verbatim}{%
  breaklines=true,breakanywhere=true,fontsize=\footnotesize,%
  breaksymbolleft={},breaksymbolright={},%
  formatcom=\rmfamily}

% Bordered, breakable container for prompts. The J/K/L appendix sections
% are wrapped in \onecolumn so each PromptBox spans the full text width;
% breakable lets long prompts (K.1) flow across pages cleanly. Padding is
% kept tight so the long prompts stay close to a single page.
\newtcolorbox{PromptBox}{%
  enhanced jigsaw, breakable,
  colback=white, colframe=black!50,
  boxrule=0.4pt, arc=1.5pt,
  left=3pt, right=3pt, top=2pt, bottom=2pt,
  before skip=4pt, after skip=4pt}
